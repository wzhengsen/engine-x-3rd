if(MSVC)
	message("MSCV detected")
	set (CMAKE_C_FLAGS "${CMAKE_C_FLAGS}")
	set (CMAKE_C_FLAGS "${CMAKE_CXX_FLAGS}")
else()
	set (CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wall -Wextra")
	set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra")
endif()

include_directories(include)
file(GLOB INCLUDES "spine-cpp/include/**/*.h")
file(GLOB SOURCES "spine-cpp/src/**/*.cpp")

add_library(ext_spine-cpp STATIC ${SOURCES} ${INCLUDES})
target_include_directories(ext_spine-cpp PUBLIC spine-cpp/include)
install(TARGETS ext_spine-cpp DESTINATION dist/lib)
install(FILES ${INCLUDES} DESTINATION dist/include)
